56 /* */
57 /* How to call Triangle from another program */
58 /* */
59 /* */
60 /* If you haven't read Triangle's instructions (run "triangle -h" to read */
61 /* them), you won't understand what follows. */
62 /* */
63 /* Triangle must be compiled into an object file (triangle.o) with the */
64 /* TRILIBRARY symbol defined (preferably by using the -DTRILIBRARY compiler */
65 /* switch). The makefile included with Triangle will do this for you if */
66 /* you run "make trilibrary". The resulting object file can be called via */
67 /* the procedure triangulate(). */
68 /* */
69 /* If the size of the object file is important to you, you may wish to */
70 /* generate a reduced version of triangle.o. The REDUCED symbol gets rid */
71 /* of all features that are primarily of research interest. Specifically, */
72 /* the -DREDUCED switch eliminates Triangle's -i, -F, -s, and -C switches. */
73 /* The CDT_ONLY symbol gets rid of all meshing algorithms above and beyond */
74 /* constrained Delaunay triangulation. Specifically, the -DCDT_ONLY switch */
75 /* eliminates Triangle's -r, -q, -a, -S, and -s switches. */
76 /* */
77 /* IMPORTANT: These definitions (TRILIBRARY, REDUCED, CDT_ONLY) must be */
78 /* made in the makefile or in triangle.c itself. Putting these definitions */
79 /* in this file will not create the desired effect. */
80 /* */
81 /* */
82 /* The calling convention for triangulate() follows. */
83 /* */
84 /* void triangulate(triswitches, in, out, vorout) */
85 /* char *triswitches; */
86 /* struct triangulateio *in; */
87 /* struct triangulateio *out; */
88 /* struct triangulateio *vorout; */
89 /* */
90 /* `triswitches' is a string containing the command line switches you wish */
91 /* to invoke. No initial dash is required. Some suggestions: */
92 /* */
93 /* - You'll probably find it convenient to use the firstnumber = 0 value */
94 /* in the TrianglateOpts structure so that points (and other items) are */
95 /* numbered from zero. This simplifies indexing, because the first item */
96 /* of any type always starts at index [0] of the corresponding array, */
97 /* whether that item's number is zero or one. */
98 /* - You'll probably want to use the `Q' (quiet) switch in your final code, */
99 /* but you can take advantage of Triangle's printed output (including the */
100 /* `V' switch) while debugging. */
101 /* - If you are not using the `q' or `a' switches, then the output points */
102 /* will be identical to the input points, except possibly for the */
103 /* boundary markers. If you don't need the boundary markers, you should */
104 /* use the `N' (no nodes output) switch to save memory. (If you do need */
105 /* boundary markers, but need to save memory, a good nasty trick is to */
106 /* set out->pointlist equal to in->pointlist before calling triangulate(),*/
107 /* so that Triangle overwrites the input points with identical copies.) */
108 /* - The `I' (no iteration numbers) and `g' (.off file output) switches */
109 /* have no effect when Triangle is compiled with TRILIBRARY defined. */
110 /* */
111 /* `in', `out', and `vorout' are descriptions of the input, the output, */
112 /* and the Voronoi output. If the `v' (Voronoi output) switch is not used, */
113 /* `vorout' may be NULL. `in' and `out' may never be NULL. */
114 /* */
115 /* Certain fields of the input and output structures must be initialized, */
116 /* as described below. */
117 /* */

120 /*****************************************************************************/
121 /* */
122 /* The `triangulateio' structure. */
123 /* */
124 /* Used to pass data into and out of the triangulate() procedure. */
125 /* */
126 /* */
127 /* Arrays are used to store points, triangles, markers, and so forth. In */
128 /* all cases, the first item in any array is stored starting at index [0]. */
129 /* However, that item is item number `1' unless the `z' switch is used, in */
130 /* which case it is item number `0'. Hence, you may find it easier to */
131 /* index points (and triangles in the neighbor list) if you use the `z' */
132 /* switch. Unless, of course, you're calling Triangle from a Fortran */
133 /* program. */
134 /* */
135 /* Description of fields (except the `numberof' fields, which are obvious): */
136 /* */
137 /* `pointlist': An array of point coordinates. The first point's x */
138 /* coordinate is at index [0] and its y coordinate at index [1], followed */
139 /* by the coordinates of the remaining points. Each point occupies two */
140 /* REALs. */
141 /* `pointattributelist': An array of point attributes. Each point's */
142 /* attributes occupy `numberofpointattributes' REALs. */
143 /* `pointmarkerlist': An array of point markers; one int per point. */
144 /* */
145 /* `trianglelist': An array of triangle corners. The first triangle's */
146 /* first corner is at index [0], followed by its other two corners in */
147 /* counterclockwise order, followed by any other nodes if the triangle */
148 /* represents a nonlinear element. Each triangle occupies */
149 /* `numberofcorners' ints. */
150 /* `triangleattributelist': An array of triangle attributes. Each */
151 /* triangle's attributes occupy `numberoftriangleattributes' REALs. */
152 /* `trianglearealist': An array of triangle area constraints; one REAL per */
153 /* triangle. Input only. */
154 /* `neighborlist': An array of triangle neighbors; three ints per */
155 /* triangle. Output only. */
156 /* */
157 /* `segmentlist': An array of segment endpoints. The first segment's */
158 /* endpoints are at indices [0] and [1], followed by the remaining */
159 /* segments. Two ints per segment. */
160 /* `segmentmarkerlist': An array of segment markers; one int per segment. */
161 /* */
162 /* `holelist': An array of holes. The first hole's x and y coordinates */
163 /* are at indices [0] and [1], followed by the remaining holes. Two */
164 /* REALs per hole. Input only, although the pointer is copied to the */
165 /* output structure for your convenience. */
166 /* */
167 /* `regionlist': An array of regional attributes and area constraints. */
168 /* The first constraint's x and y coordinates are at indices [0] and [1], */
169 /* followed by the regional attribute and index [2], followed by the */
170 /* maximum area at index [3], followed by the remaining area constraints. */
171 /* Four REALs per area constraint. Note that each regional attribute is */
172 /* used only if you select the `A' switch, and each area constraint is */
173 /* used only if you select the `a' switch (with no number following), but */
174 /* omitting one of these switches does not change the memory layout. */
175 /* Input only, although the pointer is copied to the output structure for */
176 /* your convenience. */
177 /* */
178 /* `edgelist': An array of edge endpoints. The first edge's endpoints are */
179 /* at indices [0] and [1], followed by the remaining edges. Two ints per */
180 /* edge. Output only. */
181 /* `edgemarkerlist': An array of edge markers; one int per edge. Output */
182 /* only. */
183 /* `normlist': An array of normal uni_arrays, used for infinite rays in */
184 /* Voronoi diagrams. The first normal uni_array's x and y magnitudes are */
185 /* at indices [0] and [1], followed by the remaining uni_arrays. For each */
186 /* finite edge in a Voronoi diagram, the normal vector written is the */
187 /* zero uni_array. Two REALs per edge. Output only. */
188 /* */
189 /* */
190 /* Any input fields that Triangle will examine must be initialized. */
191 /* Furthermore, for each output array that Triangle will write to, you */
192 /* must either provide space by setting the appropriate pointer to point */
193 /* to the space you want the data written to, or you must initialize the */
194 /* pointer to NULL, which tells Triangle to allocate space for the results. */
195 /* The latter option is preferable, because Triangle always knows exactly */
196 /* how much space to allocate. The former option is provided mainly for */
197 /* people who need to call Triangle from Fortran code, though it also makes */
198 /* possible some nasty space-saving tricks, like writing the output to the */
199 /* same arrays as the input. */
200 /* */
201 /* Triangle will not free() any input or output arrays, including those it */
202 /* allocates itself; that's up to you. */
203 /* */
204 /* Here's a guide to help you decide which fields you must initialize */
205 /* before you call triangulate(). */
206 /* */
207 /* `in': */
208 /* */
209 /* - `pointlist' must always point to a list of points; `numberofpoints' */
210 /* and `numberofpointattributes' must be properly set. */
211 /* `pointmarkerlist' must either be set to NULL (in which case all */
212 /* markers default to zero), or must point to a list of markers. If */
213 /* `numberofpointattributes' is not zero, `pointattributelist' must */
214 /* point to a list of point attributes. */
215 /* - If the `r' switch is used, `trianglelist' must point to a list of */
216 /* triangles, and `numberoftriangles', `numberofcorners', and */
217 /* `numberoftriangleattributes' must be properly set. If */
218 /* `numberoftriangleattributes' is not zero, `triangleattributelist' */
219 /* must point to a list of triangle attributes. If the `a' switch is */
220 /* used (with no number following), `trianglearealist' must point to a */
221 /* list of triangle area constraints. `neighborlist' may be ignored. */
222 /* - If the `p' switch is used, `segmentlist' must point to a list of */
223 /* segments, `numberofsegments' must be properly set, and */
224 /* `segmentmarkerlist' must either be set to NULL (in which case all */
225 /* markers default to zero), or must point to a list of markers. */
226 /* - If the `p' switch is used without the `r' switch, then */
227 /* `numberofholes' and `numberofregions' must be properly set. If */
228 /* `numberofholes' is not zero, `holelist' must point to a list of */
229 /* holes. If `numberofregions' is not zero, `regionlist' must point to */
230 /* a list of region constraints. */
231 /* - If the `p' switch is used, `holelist', `numberofholes', */
232 /* `regionlist', and `numberofregions' is copied to `out'. (You can */
233 /* nonetheless get away with not initializing them if the `r' switch is */
234 /* used.) */
235 /* - `edgelist', `edgemarkerlist', `normlist', and `numberofedges' may be */
236 /* ignored. */
237 /* */
238 /* `out': */
239 /* */
240 /* - `pointlist' must be initialized (NULL or pointing to memory) unless */
241 /* the `N' switch is used. `pointmarkerlist' must be initialized */
242 /* unless the `N' or `B' switch is used. If `N' is not used and */
243 /* `in->numberofpointattributes' is not zero, `pointattributelist' must */
244 /* be initialized. */
245 /* - `trianglelist' must be initialized unless the `E' switch is used. */
246 /* `neighborlist' must be initialized if the `n' switch is used. If */
247 /* the `E' switch is not used and (`in->numberofelementattributes' is */
248 /* not zero or the `A' switch is used), `elementattributelist' must be */
249 /* initialized. `trianglearealist' may be ignored. */
250 /* - `segmentlist' must be initialized if the `p' or `c' switch is used, */
251 /* and the `P' switch is not used. `segmentmarkerlist' must also be */
252 /* initialized under these circumstances unless the `B' switch is used. */
253 /* - `edgelist' must be initialized if the `e' switch is used. */
254 /* `edgemarkerlist' must be initialized if the `e' switch is used and */
255 /* the `B' switch is not. */
256 /* - `holelist', `regionlist', `normlist', and all scalars may be ignored.*/
257 /* */
258 /* `vorout' (only needed if `v' switch is used): */
259 /* */
260 /* - `pointlist' must be initialized. If `in->numberofpointattributes' */
261 /* is not zero, `pointattributelist' must be initialized. */
262 /* `pointmarkerlist' may be ignored. */
263 /* - `edgelist' and `normlist' must both be initialized. */
264 /* `edgemarkerlist' may be ignored. */
265 /* - Everything else may be ignored. */
266 /* */
267 /* After a call to triangulate(), the valid fields of `out' and `vorout' */
268 /* will depend, in an obvious way, on the choice of switches used. Note */
269 /* that when the `p' switch is used, the pointers `holelist' and */
270 /* `regionlist' are copied from `in' to `out', but no new space is */
271 /* allocated; be careful that you don't free() the same array twice. On */
272 /* the other hand, Triangle will never copy the `pointlist' pointer (or any */
273 /* others); new space is allocated for `out->pointlist', or if the `N' */
274 /* switch is used, `out->pointlist' remains uninitialized. */
275 /* */
276 /* All of the meaningful `numberof' fields will be properly set; for */
277 /* instance, `numberofedges' will represent the number of edges in the */
278 /* triangulation whether or not the edges were written. If segments are */
279 /* not used, `numberofsegments' will indicate the number of boundary edges. */
280 /* */
